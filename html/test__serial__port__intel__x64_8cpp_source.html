86 void serial_ut::test_serial_success()
87 {
88  MockRepository mocks;
89  auto intrinsics = bfn::mock_shared<intrinsics_intel_x64>(mocks);
90 
91  setup_intrinsics(mocks, intrinsics.get());
92 
93  RUN_UNITTEST_WITH_MOCKS(mocks, [&]
94  {
95  auto serial = std::make_shared<serial_port_intel_x64>(intrinsics);
96 
97  EXPECT_TRUE((g_baud_rate_lo) == ((serial_port_intel_x64::DEFAULT_BAUD_RATE & 0x00FF) >> 0));
98  EXPECT_TRUE((g_baud_rate_hi) == ((serial_port_intel_x64::DEFAULT_BAUD_RATE & 0xFF00) >> 8));
99  EXPECT_TRUE((g_interrupt_en) == 0);
100  EXPECT_TRUE((g_fifo_control & FIFO_CONTROL_ENABLE_FIFOS) != 0);
101  EXPECT_TRUE((g_fifo_control & FIFO_CONTROL_CLEAR_RECIEVE_FIFO) != 0);
102  EXPECT_TRUE((g_fifo_control & FIFO_CONTROL_CLEAR_TRANSMIT_FIFO) != 0);
103  EXPECT_TRUE((g_line_control & LINE_CONTROL_DATA_MASK) == serial_port_intel_x64::DEFAULT_DATA_BITS);
104  EXPECT_TRUE((g_line_control & LINE_CONTROL_STOP_MASK) == serial_port_intel_x64::DEFAULT_STOP_BITS);
105  EXPECT_TRUE((g_line_control & LINE_CONTROL_PARITY_MASK) == serial_port_intel_x64::DEFAULT_PARITY_BITS);
106 
107  EXPECT_TRUE(serial->baud_rate() == serial_port_intel_x64::DEFAULT_BAUD_RATE);
108  EXPECT_TRUE(serial->data_bits() == serial_port_intel_x64::DEFAULT_DATA_BITS);
109  EXPECT_TRUE(serial->stop_bits() == serial_port_intel_x64::DEFAULT_STOP_BITS);
110  EXPECT_TRUE(serial->parity_bits() == serial_port_intel_x64::DEFAULT_PARITY_BITS);
111  });
