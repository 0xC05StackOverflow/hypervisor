91 void
92 vmxon_intel_x64::check_vmx_capabilities_msr()
93 {
94  auto vmx_basic_msr = m_intrinsics->read_msr(IA32_VMX_BASIC_MSR);
95  auto physical_address_width = (vmx_basic_msr >> 48) & 0x1;
96  auto memory_type = (vmx_basic_msr >> 50) & 0xF;
97 
98  if (physical_address_width != 0)
99  throw vmxon_capabilities_failure(
100  vmx_basic_msr, physical_address_width);
101 
102  if (memory_type != 6)
103  throw vmxon_capabilities_failure(
104  vmx_basic_msr, memory_type);
105 
106  if ((vmx_basic_msr & (1ULL << 55)) == 0)
107  throw vmxon_capabilities_failure(
108  vmx_basic_msr, 55);
109 }

111 void
112 vmxon_intel_x64::check_ia32_vmx_cr0_fixed_msr()
113 {
114  auto cr0 = m_intrinsics->read_cr0();
115  auto ia32_vmx_cr0_fixed0 = m_intrinsics->read_msr(IA32_VMX_CR0_FIXED0_MSR);
116  auto ia32_vmx_cr0_fixed1 = m_intrinsics->read_msr(IA32_VMX_CR0_FIXED1_MSR);
117 
118  if (0 != ((~cr0 & ia32_vmx_cr0_fixed0) | (cr0 & ~ia32_vmx_cr0_fixed1)))
119  throw vmxon_fixed_msr_failure(
120  cr0, ia32_vmx_cr0_fixed0, ia32_vmx_cr0_fixed1);

166 vmxon_intel_x64::create_vmxon_region()
167 {
168  auto cor1 = commit_or_rollback([&]

170 
171  auto region = (uint32_t *)g_mm->malloc_aligned(4096, 4096);
172 
173  m_vmxon_region = std::unique_ptr<uint32_t>(region);
174  m_vmxon_region_phys = (uintptr_t)g_mm->virt_to_phys(region);
175
