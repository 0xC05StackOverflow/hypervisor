59 vmcs_intel_x64::check_control_pin_based_ctls_reserved_properly_set()
60 {
61  auto ia32_vmx_pinbased_ctls_msr =
62  m_intrinsics->read_msr(IA32_VMX_TRUE_PINBASED_CTLS_MSR);
63 
64  auto allowed_zero = ((ia32_vmx_pinbased_ctls_msr >> 00) & 0x00000000FFFFFFFF);
65  auto allowed_one = ((ia32_vmx_pinbased_ctls_msr >> 32) & 0x00000000FFFFFFFF);
66 
67  auto ctls = get_pin_ctls();
68  auto ctls_lower = ((ctls >> 00) & 0x00000000FFFFFFFF);
69  auto ctls_upper = ((ctls >> 32) & 0x00000000FFFFFFFF);
70 
71  if ((allowed_zero & ctls_lower) != allowed_zero || (allowed_one & ~ctls_upper) != allowed_one)
72  throw vmcs_invalid_ctls("pin based", allowed_zero, allowed_one,
73  ctls_lower, ctls_upper);

158 void
159 vmcs_intel_x64::check_control_tpr_shadow_and_virtual_apic()
160 {
161  if (is_enabled_tpr_shadow())
162  {
163  auto phys_addr = vmread(VMCS_VIRTUAL_APIC_ADDRESS_FULL);
164 
165  if (phys_addr == 0)
166  throw invalid_address("virtual apic physical addr is NULL", phys_addr);
167 
168  if ((phys_addr & 0x0000000000000FFF) != 0)
169  throw invalid_alignment("virtual apic addr not 4k aligned", phys_addr);
170 
171  if (is_physical_address_valid(phys_addr) == false)
172  throw invalid_address("virtual apic addr too large", phys_addr);
173 
174  if (is_enabled_virtual_interrupt_delivery() == false)
175  throw hardware_unsupported("virtual interrupt delivery is disabled or unsupported");
176 
177  auto tpr_threshold = vmread(VMCS_TPR_THRESHOLD);
178 
179  if ((tpr_threshold & 0x00000000FFFFFFF0) != 0)
180  throw vmcs_invalid_field("bits 31:4 must be 0", tpr_threshold);
181 
182  if (is_enabled_virtualized_apic() == false)
183  throw hardware_unsupported("virtualized apics are disabled or unsupported");
184 
185  auto virt_addr = (uint64_t)g_mm->phys_to_virt((void *)phys_addr);
186 
187  if (virt_addr == 0)
188  throw invalid_address("vitual apic virtual addr is NULL", virt_addr);
189 
190  // This is described in 26.2.1.1
191  auto vtpr = (((uint8_t *)((uint8_t *)virt_addr + 0x80))[0]);
192  auto vtpr_74 = (vtpr & 0xF0) >> 4;
193  auto tpr_threshold_30 = (uint8_t)(tpr_threshold & 0x000000000000000F);
194 
195  if (tpr_threshold_30 > vtpr_74)
196  throw vmcs_invalid_field("invalid TPR threshold", tpr_threshold);
197  }
198  else
199  {
200  if (is_enabled_x2apic_mode())
201  throw hardware_unsupported("x2apic mode must be disabled if tpr shadow is disabled");
202 
203  if (is_enabled_apic_register_virtualization())
204  throw hardware_unsupported("apic register virtualization must be disabled if tpr shadow is disabled");
205 
206  if (is_enabled_virtual_interrupt_delivery())
207  throw hardware_unsupported("virtual interrupt delivery must be disabled if tpr shadow is disabled");
208  }

313 void
314 vmcs_intel_x64::check_control_enable_ept_checks()
315 {
316  if (is_enabled_ept() == false)
317  return;
318 
319  auto eptp = vmread(VMCS_EPT_POINTER_FULL);
320 
321  auto ia32_vmx_ept_vpid_cap_msr =
322  m_intrinsics->read_msr(IA32_VMX_EPT_VPID_CAP_MSR);
323 
324  auto uncacheable = (ia32_vmx_ept_vpid_cap_msr & IA32_VMX_EPT_VPID_CAP_UC);
325  auto write_back = (ia32_vmx_ept_vpid_cap_msr & IA32_VMX_EPT_VPID_CAP_WB);
326 
327  if ((eptp & EPTP_MEMORY_TYPE) == 0 && uncacheable == 0)
328  throw vmcs_invalid_field("hardware does not support ept memory type: "
329  "uncachable", ia32_vmx_ept_vpid_cap_msr);
330 
331  if ((eptp & EPTP_MEMORY_TYPE) == 6 && write_back == 0)
332  throw vmcs_invalid_field("hardware does not support ept memory type: "
333  "write-back", ia32_vmx_ept_vpid_cap_msr);
334 
335  if ((eptp & EPTP_MEMORY_TYPE) != 0 && (eptp & EPTP_MEMORY_TYPE) != 6)
336  throw vmcs_invalid_field("unknown eptp memory type", eptp);
337 
338  if ((eptp & EPTP_PAGE_WALK_LENGTH) >> 3 != 3)
339  throw vmcs_invalid_field("the ept walk-through length must 1 less "
340  "than 4, i.e. 3", eptp);
341 
342  auto ad = (ia32_vmx_ept_vpid_cap_msr & IA32_VMX_EPT_VPID_CAP_AD);
343 
344  if ((eptp & EPTP_ACCESSED_DIRTY_FLAGS_ENABLED) != 0 && ad == 0)
345  throw vmcs_invalid_field("hardware does not support dirty / "
346  "accessed flags for ept",
347  ia32_vmx_ept_vpid_cap_msr);
348 
349  if ((eptp & 0xFFFF000000000000) != 0 || (eptp & 0x0000000000000F80) != 0)
350  throw vmcs_invalid_field("bits 11:7 and 63:48 must be 0", eptp);
351 }

569  vmread(VMCS_VM_ENTRY_INTERRUPTION_INFORMATION_FIELD);
570
