36 void
37 vmcs_intel_x64::launch(const vmcs_intel_x64_state &host_state,
38  const vmcs_intel_x64_state &guest_state)
39 {
40  auto cor1 = commit_or_rollback([&]
41  {
42  this->release_vmcs_region();
43  this->release_exit_handler_stack();
44  });
45 
46  this->create_vmcs_region();
47  this->create_exit_handler_stack();
48 
49  if (m_intrinsics->vmclear(&m_vmcs_region_phys) == false)
50  throw vmcs_failure("failed to clear vmcs");
51 
52  if (m_intrinsics->vmptrld(&m_vmcs_region_phys) == false)
53  throw vmcs_failure("failed to load vmcs");
54 
55  this->write_16bit_guest_state(guest_state);
56  this->write_64bit_guest_state(guest_state);
57  this->write_32bit_guest_state(guest_state);
58  this->write_natural_guest_state(guest_state);
59 
60  this->write_16bit_control_state(host_state);
61  this->write_64bit_control_state(host_state);
62  this->write_32bit_control_state(host_state);
63  this->write_natural_control_state(host_state);
64 
65  this->write_16bit_host_state(host_state);
66  this->write_64bit_host_state(host_state);
67  this->write_32bit_host_state(host_state);
68  this->write_natural_host_state(host_state);
69 
70  this->pin_based_vm_execution_controls();
71  this->primary_processor_based_vm_execution_controls();
72  this->secondary_processor_based_vm_execution_controls();
73  this->vm_exit_controls();
74  this->vm_entry_controls();
75 
76  if (m_intrinsics->vmlaunch() == false)
77  {
78  this->dump_vmcs();
79 
80  this->print_execution_controls();
81  this->print_pin_based_vm_execution_controls();
82  this->print_primary_processor_based_vm_execution_controls();
83  this->print_secondary_processor_based_vm_execution_controls();
84  this->print_vm_exit_control_fields();
85  this->print_vm_entry_control_fields();
86 
87  host_state.dump("Host");
88  guest_state.dump("Guest");
89 
90  this->check_vmcs_control_state();
91  // this->check_vmcs_guest_state();
92  // this->check_vmcs_host_state();
93 
94  throw vmcs_launch_failure(this->get_vm_instruction_error());
95  }
96 
97  cor1.commit();
98 }

117 void
118 vmcs_intel_x64::create_vmcs_region()
119 {
120  auto cor1 = commit_or_rollback([&]
121  { this->release_vmcs_region(); });
122 
123  auto region = (uint32_t *)g_mm->malloc_aligned(4096, 4096);
124 
125  m_vmcs_region = std::unique_ptr<uint32_t>(region);
126  m_vmcs_region_phys = (uintptr_t)g_mm->virt_to_phys(region);
127 
128  region[0] = m_intrinsics->read_msr(IA32_VMX_BASIC_MSR) & 0x7FFFFFFFF;
129 
130  cor1.commit();

199 void
200 vmcs_intel_x64::write_32bit_control_state(const vmcs_intel_x64_state &state)
201 {
202  (void) state;
203 
204  uint64_t lower;
205  uint64_t upper;
206 
207  auto ia32_vmx_pinbased_ctls_msr =
208  m_intrinsics->read_msr(IA32_VMX_TRUE_PINBASED_CTLS_MSR);
209  auto ia32_vmx_procbased_ctls_msr =
210  m_intrinsics->read_msr(IA32_VMX_TRUE_PROCBASED_CTLS_MSR);
211  auto ia32_vmx_exit_ctls_msr =
212  m_intrinsics->read_msr(IA32_VMX_TRUE_EXIT_CTLS_MSR);
213  auto ia32_vmx_entry_ctls_msr =
214  m_intrinsics->read_msr(IA32_VMX_TRUE_ENTRY_CTLS_MSR);
215 
216  lower = ((ia32_vmx_pinbased_ctls_msr >> 0) & 0x00000000FFFFFFFF);
217  upper = ((ia32_vmx_pinbased_ctls_msr >> 32) & 0x00000000FFFFFFFF);
218  vmwrite(VMCS_PIN_BASED_VM_EXECUTION_CONTROLS, lower & upper);
219 
220  lower = ((ia32_vmx_procbased_ctls_msr >> 0) & 0x00000000FFFFFFFF);
221  upper = ((ia32_vmx_procbased_ctls_msr >> 32) & 0x00000000FFFFFFFF);
222  vmwrite(VMCS_PRIMARY_PROCESSOR_BASED_VM_EXECUTION_CONTROLS, lower & upper);
223 
224  lower = ((ia32_vmx_exit_ctls_msr >> 0) & 0x00000000FFFFFFFF);
225  upper = ((ia32_vmx_exit_ctls_msr >> 32) & 0x00000000FFFFFFFF);
226  vmwrite(VMCS_VM_EXIT_CONTROLS, lower & upper);
227 
228  lower = ((ia32_vmx_entry_ctls_msr >> 0) & 0x00000000FFFFFFFF);
229  upper = ((ia32_vmx_entry_ctls_msr >> 32) & 0x00000000FFFFFFFF);
230  vmwrite(VMCS_VM_ENTRY_CONTROLS, lower & upper);
231 
232  // unused: VMCS_EXCEPTION_BITMAP
233  // unused: VMCS_PAGE_FAULT_ERROR_CODE_MASK
234  // unused: VMCS_PAGE_FAULT_ERROR_CODE_MATCH
