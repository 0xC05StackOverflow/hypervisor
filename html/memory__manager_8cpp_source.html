42 // The following defines the entire memory pool. This pool will be used by the
43 // VMM to create the different resources that it needs, which is mainly given
44 // out using new/delete. If space runs out, this will need to be increased.
45 uint8_t g_mem_pool[MAX_MEM_POOL] ALIGN_MEMORY = {0};
46 
47 // The memory pool itself is given out in blocks. Any attempt to new / delete
48 // will always allocate at least a block of memory, which is usually set to a
49 // cache line size. This stores whether a block is allocated or not. Note that
50 // the way we know if a block is allocated, is it's not set to FREE_BLOCK.
51 // We use the value stored here to tell us what the starting address is when a
52 // block is allocated. This greatly simplifies freeing memory at the expense of
53 // a lot of memory needed to manage memory.
54 int64_t g_block_allocated[MAX_BLOCKS] = {0};

138 void *
139 memory_manager::malloc_aligned(size_t size, uint64_t alignment)
140 {
141  if (size == 0)
142  return 0;
143 
144  // This is a really simple "first fit" algorithm. The only optimization
145  // that we have here is m_start. This algorithm works by looping from the
146  // start of the list of blocks, and looking for a contiguous set of
147  // blocks, that matches the provided alignment. Once we find the set of
148  // blocks the user is asking for, we set each block to "allocated" by
149  // providing the start block for the chunk of memory that was allocated.
150  // Free will use this "start" block to identify the starting block for
151  // any allocated virtual address.
152  //
153  // m_start defines the starting position to search the list of blocks.
154  // Without m_start, we would start from 0, and loop to MAX_BLOCKS on
155  // every allocation. In practice, we don't need to start from 0, as each
156  // block is likely to be consumed as we allocate more and more memory.
157  // It's not until the first hole or "fragmentation" occurs, that m_start
158  // must stop until the fragmentation is removed.
159 
160  auto count = 0U;
161  auto block = 0U;
162  auto reset = 0U;
163  auto num_blocks = size >> MAX_CACHE_LINE_SHIFT;
164 
165  if ((size & (MAX_CACHE_LINE_SIZE - 1)) != 0)
166  num_blocks++;
167 
168  for (auto b = m_start; b < MAX_BLOCKS && count < num_blocks; b++)
169  {
170  if (g_block_allocated[b] == FREE_BLOCK)
171  {
172  if (count == 0)
173  {
174  if (is_block_aligned(b, alignment) == false)
175  {
176  reset = 1;
177  continue;
178  }
179 
180  block = b;
181  }
182 
183  count++;
184 
185  if (reset == 0 && b > m_start)
186  m_start = b;
187  }
188  else
189  {
190  if (count > 0)
191  reset = 1;
192 
193  count = 0;
194  }
195  }
196 
197  if (count == num_blocks)
198  {
199  for (auto b = block; b < MAX_BLOCKS && b < num_blocks + block; b++)
200  g_block_allocated[b] = block;
201 
202  return block_to_virt(block);
203  }
204 
205  return out_of_memory();
206 }

208 void
209 memory_manager::free(void *ptr)
210 {
211  if (ptr == 0)
212  return;
213 
214  // Our version of free is a lot cleaner than most memory manager, but is
215  // terribly inefficent with respect to how much memory it consumes for
216  // bookeeping. We store the starting block for every virtual address. This
217  // means that if you were to delete a base class for a subclass that did
218  // not specify "virtual" for the base class, all of memory would still be
219  // deleted. This is because we know where the starting address should be
220  // even if the virtual address that was provided is offset from the
221  // virtual address that was actually allocated.
222  //
223  // Also note that we need to adjust m_start if we freed memory that opened
224  // up a "fragmentation" in list of allocated blocks.
225 
226  auto block = virt_to_block(ptr);
227 
228  if (block < 0 || (uint32_t)block >= MAX_BLOCKS)
229  return;
230 
231  auto start = g_block_allocated[block];
232 
233  if (start < 0 || (uint32_t)start >= MAX_BLOCKS)
234  return;
235 
236  for (auto b = (uint32_t)start; b < MAX_BLOCKS; b++)
237  {
238  if (b < m_start)
239  m_start = b;
240 
241  if (g_block_allocated[b] != start)
242  break;
243 
244  g_block_allocated[b] = FREE_BLOCK;
245  }
246 }

267 memory_manager::virt_to_phys(void *virt)
268 {
269  auto key = (uintptr_t)virt >> MAX_PAGE_SHIFT;
270  const auto &md_iter = m_virt_to_phys_map.find(key);
271 
272  if (md_iter == m_virt_to_phys_map.end())
273  return 0;
274

304 memory_manager::is_block_aligned(int64_t block, int64_t alignment)
305 {
306  if (block < 0 || (uint32_t)block >= MAX_BLOCKS)
307  return false;
308

318 void
319 memory_manager::add_mdl(memory_descriptor *mdl, int64_t num)
320 {
321  if (mdl == NULL)
322  throw std::invalid_argument("mdl == NULL");
323 
324  if (num == 0)
325  throw std::invalid_argument("num == 0");
326 
327  for (auto i = 0; i < num; i++)
328  {
329  const auto &md = mdl[i];
330 
331  if (md.size != MAX_PAGE_SIZE)
332  throw invalid_mdl("md.size != MAX_PAGE_SIZE", i);
333 
334  if (((uintptr_t)md.virt & (MAX_PAGE_SIZE - 1)) != 0)
335  throw invalid_mdl("virt address is not page aligned", i);
336 
337  if (((uintptr_t)md.phys & (MAX_PAGE_SIZE - 1)) != 0)
338  throw invalid_mdl("phys address is not page aligned", i);
339 
340  auto cor1 = commit_or_rollback([&]
341  {
342  m_virt_to_phys_map.erase((uintptr_t)md.virt >> MAX_PAGE_SHIFT);
343  m_phys_to_virt_map.erase((uintptr_t)md.phys >> MAX_PAGE_SHIFT);
344  });
345 
346  m_virt_to_phys_map[(uintptr_t)md.virt >> MAX_PAGE_SHIFT] = md;
347  m_phys_to_virt_map[(uintptr_t)md.phys >> MAX_PAGE_SHIFT] = md;
348 
349  cor1.commit();
350  }
351 }
