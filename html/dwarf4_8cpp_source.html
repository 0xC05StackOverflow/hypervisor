198 static uint64_t
199 private_decode_reg(const cfi_register &reg, uint64_t cfa, register_state *state)
200 {
201  uint64_t value;
202 
203  log("getting r%ld (%s) ", reg.index(), state->name(reg.index()));
204 
205  switch (reg.rule())
206  {
207  case rule_undefined:
208  ABORT("unable to get register value for unused register");
209 
210  case rule_same_value:
211  log("from r%ld\n", reg.index());
212  value = state->get(reg.index());
213  break;
214 
215  case rule_offsetn:
216  log("from cfa(0x%08lx) + n(%ld)\n", cfa, (int64_t)reg.index());
217  value = ((uint64_t *)(cfa + (int64_t)reg.value()))[0];
218  break;
219 
220  case rule_val_offsetn:
221  log("from val cfa(0x%08lx) + n(%ld)\n", cfa, (int64_t)reg.index());
222  value = cfa + (int64_t)reg.value();
223  break;
224 
225  case rule_register:
226  log("from r%ld\n", reg.value());
227  value = state->get(reg.value());
228  break;
229 
230  case rule_expression:
231  ABORT("DWARF4 expressions currently not supported");
232 
233  case rule_val_expression:
234  ABORT("DWARF4 expressions currently not supported");
235 
236  default:
237  ABORT("unknown rule. cfi table is malformed");
238  }
239 
240  log(" - value: 0x%08lx\n", value);
241 
242  return value;
243 }

486 static void
487 private_parse_instructions(cfi_table_row *row,
488  const ci_entry &cie,
489  const fd_entry &fde,
490  register_state *state,
491  bool is_cie)
492 {
493  uint64_t pc_begin = is_cie ? 0 : fde.pc_begin();
494  uint64_t l1 = is_cie ? 0ULL : state->get_ip() - fde.pc_begin();
495  uint64_t l2 = 0ULL;
496 
497  char *p = is_cie ? cie.initial_instructions() : fde.instructions();
498  char *end = is_cie ? cie.entry_end() : fde.entry_end();
499 
500  while (p < end && l1 >= l2)
501  private_parse_instruction(row, cie, &p, &l1, &l2, pc_begin, state);
502 }

533  log("%08lx ", fde_offset2);

550 int64_t
551 dwarf4::decode_sleb128(char **addr)
552 {
553  uint8_t byte = 0;
554  int64_t shift = 0;
555  int64_t result = 0;
556 
557  while (1)
558  {
559  byte = *((uint8_t *)(*addr)++);
560  result |= ((byte & 0x7f) << shift);
561  shift += 7;
562  if ((byte & 0x80) == 0)
563  break;
564  }
565 
566  if ((byte & 0x40) != 0)
567  result |= (-1LL) << shift;
568 
569  return result;
570 }

591 void
592 dwarf4::unwind(const fd_entry &fde, register_state *state)
593 {
594  log("-------------\n");
595  log("- Unwinding -\n");
596  log("-------------\n");
597  log("\n");
598 
599  auto row = private_decode_cfi(fde, state);
600  auto cfa = private_decode_cfa(row, state);
601 
602  for (auto i = 0U; i < state->max_num_registers(); i++)
603  {
604  auto reg = row.reg(i);
605 
606  if (reg.rule() == rule_undefined)
607  continue;
608 
609  state->set(i, private_decode_reg(reg, cfa, state));
610  }
611 
612  state->commit(cfa + row.arg_size());
613 }
