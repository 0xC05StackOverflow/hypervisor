25 std::string
26 vmcs_intel_x64::get_vm_instruction_error()
27 {
28  switch (vmread(VMCS_VM_INSTRUCTION_ERROR))

30  case 1:
31  return "VMCALL executed in VMX root operation";
32 
33  case 2:
34  return "VMCLEAR with invalid physical address";
35 
36  case 3:
37  return "VMCLEAR with VMXON pointer";
38 
39  case 4:
40  return "VMLAUNCH with non-clear VMCS";
41 
42  case 5:
43  return "VMRESUME with non-launched VMCS";
44 
45  case 6:
46  return "VMRESUME after VMXOFF (VMXOFF and VMXON between "
47  "VMLAUNCH and VMRESUME)";
48 
49  case 7:
50  return "VM entry with invalid control field(s)";
51 
52  case 8:
53  return "VM entry with invalid host-state field(s)";
54 
55  case 9:
56  return "VMPTRLD with invalid physical address";
57 
58  case 10:
59  return "VMPTRLD with VMXON pointer";
60 
61  case 11:
62  return "VMPTRLD with incorrect VMCS revision identifier";
63 
64  case 12:
65  return "VMREAD/VMWRITE from/to unsupported VMCS component";
66 
67  case 13:
68  return "VMWRITE to read-only VMCS component";
69 
70  case 15:
71  return "VMXON executed in VMX root operation";
72 
73  case 16:
74  return "VM entry with invalid executive-VMCS pointer";
75 
76  case 17:
77  return "VM entry with non-launched executive VMCS";
78 
79  case 18:
80  return "VM entry with executive-VMCS pointer not VMXON "
81  "pointer (when attempting to deactivate the "
82  "dual-monitor treatment of SMIs and SMM)";
83 
84  case 19:
85  return "VMCALL with non-clear VMCS (when attempting to "
86  "activate the dual-monitor treatment of SMIs and "
87  "SMM)";
88 
89  case 20:
90  return "VMCALL with invalid VM-exit control fields";
91 
92  case 22:
93  return "VMCALL with incorrect MSEG revision identifier "
94  "(when attempting to activate the dual-monitor "

96 
97  case 23:
98  return "VMXOFF under dual-monitor treatment of SMIs and "
99  "SMM";
100 
101  case 24:
102  return "VMCALL with invalid SMM-monitor features (when "
103  "attempting to activate the dual-monitor treatment "
104  "of SMIs and SMM)";
105 
106  case 25:
107  return "VM entry with invalid VM-execution control fields "
108  "in executive VMCS (when attempting to return from "
109  "SMM)";
110 
111  case 26:
112  return "VM entry with events blocked by MOV SS.";
113 
114  case 28:
115  return "Invalid operand to INVEPT/INVVPID.";
116 
117  default:
118  return "Unknown VM instruction error";
119  }
