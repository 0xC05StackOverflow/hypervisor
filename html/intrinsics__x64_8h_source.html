297  virtual uint64_t
298  segment_descriptor(uint16_t selector) const noexcept
299  {
300  gdt_t gdt_reg;
301  read_gdt(&gdt_reg);
302 
303  // Intel defines a null selector as a selector = 0. Basically, the
304  // first entry in the GDT is not used, and the processor views the
305  // 0 index as a "null selector". In 32bit mode, this would cause a
306  // GP fault if you attempted to use this type of selector. In 64bit
307  // mode, most of the selectors are null.
308  if (selector == 0)
309  return 0;
310 
311  // The global descriptor table is a table of segement descriptors.
312  // The decriptors can take on different forms, and their definition
313  // is in the intel's software developer's manual, volume 3, chapter
314  // 3.5. In 64bit mode, code / data segments are still 32bits, while
315  // TSS descriptors (defined in chapter 7.2.3), states that a
316  // TSS descriptor is actually 16 bytes long, as more space is needed
317  // for the 64bit base address.In general, even though there are
318  // different types of descriptors, they all take on a similar form,
319  // which is best described in chapter 3.4.5.
320  uint64_t *gdt = (uint64_t *)gdt_reg.base;
321 
322  // Each selector is 16bits, with bit 1-0 indicating the requested
323  // privilege level (RPL) and bit 2 indicating the table indicator (TI).
324  // The index into the GDT is the remaining bits, thus, to get to the
325  // index we remove the RPL and TI bit.
326  selector = (selector >> 3);
327 
328  // Finally, return the 8-byte segment descriptor
329  return gdt[selector];
330  }

332  virtual uint32_t
333  segment_descriptor_limit(uint16_t selector) const noexcept
334  {
335  // The segment limit description can be found in the intel's software
336  // developer's manual, volume 3, chapter 3.4.5 as well as volume 3,
337  // chapter 24.4.1.
338  //
339  // ------------------------------------------------------------------
340  // | | Limit 19-16 | |
341  // ------------------------------------------------------------------
342  // | | Limit 15-00 |
343  // ------------------------------------------------------------------
344  //
345  // Note that for the limit, we use LSL because it translates the limit
346  // field for us. If the granularity bit in the segment descriptor is
347  // set to 1, the limit is measured in pages, while if the granularity
348  // bit is set to 0, it's measured in bytes. a VMCS however only takes
349  // bytes, so we return the limit from LSL as it will do this
350  // translation for us.
351 
352  return load_segment_limit(selector);
353  }

355  virtual uint64_t
356  segment_descriptor_base(uint16_t selector) const noexcept
357  {
358  uint64_t sd1 = segment_descriptor(selector);
359  uint64_t sd2 = segment_descriptor(selector + (1 << 3));
360  uint64_t base_15_00 = ((sd1 & 0x00000000FFFF0000) >> 16);
361  uint64_t base_23_16 = ((sd1 & 0x000000FF00000000) >> 16);
362  uint64_t base_31_24 = ((sd1 & 0xFF00000000000000) >> 32);
363  uint64_t base_63_32 = ((sd2 & 0x00000000FFFFFFFF) << 32);
364 
365  // If we have a null selector, we return 0 since this is not a valid
366  // selector into the GDT
367  if (selector == 0)
368  return 0;
369 
370  // The segment base description can be found in the intel's software
371  // developer's manual, volume 3, chapter 3.4.5 as well as volume 3,
372  // chapter 24.4.1.
373  //
374  // Note that in 64bit mode, system descriptors are 16 bytes long
375  // instread of the traditional 8 bytes. A system descriptor has the
376  // system flag set to 0. Most of the time, this is going to be the
377  // TSS descriptor. Even though Intel Tasks don't exist in 64 bit mode,
378  // the TSS descriptor is still used, and thus, TR must still be loaded.
379  //
380  // Note that the selector index starts in bit 3 in the segment
381  // selector register, so to add 1 to the register, you are actually
382  // adding 0x8, or (1 << 3) to get the next descriptor in the GDT.
383  //
384  // ------------------------------------------------------------------
385  // | Base 63-32 |
386  // ------------------------------------------------------------------
387  // | Base 31-24 | | Base 23-16 |
388  // ------------------------------------------------------------------
389  // | Base 15-00 | |
390  // ------------------------------------------------------------------
391  //
392 
393  if ((sd1 & 0x100000000000) == 0)
394  {
395  return base_63_32 | base_31_24 | base_23_16 | base_15_00;
396  }
397  else
398  {
399  return base_31_24 | base_23_16 | base_15_00;
400  }
401  }

403  virtual uint32_t
404  segment_descriptor_access(uint16_t selector) const noexcept
405  {
406  uint64_t sd = segment_descriptor(selector);
407  uint64_t access_07_00 = ((sd & 0x0000FF0000000000) >> 40);
408  uint64_t access_15_12 = ((sd & 0x00F0000000000000) >> 40);
409 
410  // Note that the Intel manual, in chapter 24.4.1, states that there is
411  // an extra "usable" bit that we need to account for. Specifically they
412  // state: "Bit 16 indicates an unusable segment. Attempts to use such
413  // a segment fault except in 64-bit mode. In general, a segment
414  // register is unusable if it has been loaded with a null selector."
415  //
416  if (selector == 0)
417  return 0x10000;
418 
419  // The segment access description can be found in the intel's software
420  // developer's manual, volume 3, chapter 3.4.5 as well as volume 3,
421  // chapter 24.4.1.
422  //
423  // ------------------------------------------------------------------
424  // | | A 15-12 | | Access 07-00 | |
425  // ------------------------------------------------------------------
426  // | | |
427  // ------------------------------------------------------------------
428  //
429 
430  return access_15_12 | access_07_00;
431  }
