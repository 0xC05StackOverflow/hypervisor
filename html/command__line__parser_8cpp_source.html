40 command_line_parser::parse(const std::vector<std::string> &args)
41 {
42  for (const auto &arg : args)
43  {
44  if (arg != "-h" && arg != "--help")
45  continue;
46 
47  return reset();
48  }
49 
50  for (auto i = 0U; i < args.size(); i++)
51  {
52  const auto &arg = args[i];
53 
54  if (arg.empty() == true)
55  continue;
56 
57  if (arg[0] == '-')
58  continue;
59 
60  if (arg == "load") return parse_load(args, i);
61  if (arg == "unload") return parse_unload(args, i);
62  if (arg == "start") return parse_start(args, i);
63  if (arg == "stop") return parse_stop(args, i);
64  if (arg == "dump") return parse_dump(args, i);
65  if (arg == "status") return parse_status(args, i);
66 
67  throw unknown_command(arg);
68  }
69 
70  return reset();
71 }
