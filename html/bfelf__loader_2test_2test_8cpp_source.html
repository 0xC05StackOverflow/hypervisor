35 bool bfelf_loader_ut::init()
36 {
37  std::ifstream dummy_misc_ifs(c_dummy_misc_filename, std::ifstream::ate);
38  std::ifstream dummy_code_ifs(c_dummy_code_filename, std::ifstream::ate);
39 
40  auto cor1 = commit_or_rollback([&]
41  {
42  dummy_misc_ifs.close();
43  dummy_code_ifs.close();
44  });
45 
46  if (dummy_misc_ifs.is_open() == false ||
47  dummy_code_ifs.is_open() == false)
48  {
49  std::cout << "unable to open one or more dummy libraries: " << std::endl;
50  std::cout << " - dummy_misc: " << dummy_misc_ifs.is_open() << std::endl;
51  std::cout << " - dummy_code: " << dummy_code_ifs.is_open() << std::endl;
52  return false;
53  }
54 
55  m_dummy_misc_length = dummy_misc_ifs.tellg();
56  m_dummy_code_length = dummy_code_ifs.tellg();
57 
58  if (m_dummy_misc_length == 0 ||
59  m_dummy_code_length == 0)
60  {
61  std::cout << "one or more of the dummy libraries is empty: " << std::endl;
62  std::cout << " - dummy_misc: " << m_dummy_misc_length << std::endl;
63  std::cout << " - dummy_code: " << m_dummy_code_length << std::endl;
64  return false;
65  }
66 
67  m_dummy_misc = std::shared_ptr<char>(new char[m_dummy_misc_length]());
68  m_dummy_code = std::shared_ptr<char>(new char[m_dummy_code_length]());
69 
70  auto cor2 = commit_or_rollback([&]
71  {
72  m_dummy_misc.reset();
73  m_dummy_code.reset();
74  });
75 
76  if (!m_dummy_misc ||
77  !m_dummy_code)
78  {
79  std::cout << "unable to allocate space for one or more of the dummy libraries: " << std::endl;
80  std::cout << " - dummy_misc: " << (void *)m_dummy_misc.get() << std::endl;
81  std::cout << " - dummy_code: " << (void *)m_dummy_code.get() << std::endl;
82  return false;
83  }
84 
85  dummy_misc_ifs.seekg(0);
86  dummy_code_ifs.seekg(0);
87 
88  dummy_misc_ifs.read(m_dummy_misc.get(), m_dummy_misc_length);
89  dummy_code_ifs.read(m_dummy_code.get(), m_dummy_code_length);
90 
91  if (dummy_misc_ifs.fail() == true ||
92  dummy_code_ifs.fail() == true)
93  {
94  std::cout << "unable to load one or more dummy libraries into memory: " << std::endl;
95  std::cout << " - dummy_misc: " << dummy_misc_ifs.fail() << std::endl;
96  std::cout << " - dummy_code: " << dummy_code_ifs.fail() << std::endl;
97  return false;
98  }
99 
100  cor1.commit();
101  cor2.commit();
102 
103  return true;
104 }

230 std::shared_ptr<char>
231 bfelf_loader_ut::load_elf_file(bfelf_file_t *ef)
232 {
233  bfelf64_xword total = 0;
234  bfelf64_xword num_segments = bfelf_file_num_segments(ef);
235 
236  for (auto i = 0U; i < num_segments; i++)
237  {
238  auto ret = 0;
239  bfelf_phdr *phdr = 0;
240 
241  ret = bfelf_file_get_segment(ef, i, &phdr);
242  if (ret != BFELF_SUCCESS)
243  return false;
244 
245  if (total < phdr->p_vaddr + phdr->p_memsz)
246  total = phdr->p_vaddr + phdr->p_memsz;
247  }
248 
249  auto exec = (char *)alloc_exec(total);
250 
251  if (!exec)
252  return 0;
253 
254  memset(exec, 0, total);
255 
256  for (auto i = 0U; i < num_segments; i++)
257  {
258  auto ret = 0;
259  bfelf_phdr *phdr = 0;
260 
261  ret = bfelf_file_get_segment(ef, i, &phdr);
262  if (ret != BFELF_SUCCESS)
263  return false;
264 
265  auto exec_p = exec + phdr->p_vaddr;
266  auto file_p = ef->file + phdr->p_offset;
267 
268  memcpy(exec_p, file_p, phdr->p_filesz);
269  }
270 
271  return std::shared_ptr<char>(exec, [](void *) {});
272 }
