167 static bfelf64_sword
168 private_elf_string_equals(struct e_string_t *str1, struct e_string_t *str2)
169 {
170  bfelf64_sword i = 0;
171 
172  if (str1->len != str2->len)
173  return BFELF_ERROR_MISMATCH;
174 
175  if (str1->len == 0)
176  return BFELF_ERROR_MISMATCH;
177 
178  for (i = 0; i < str1->len; i++)
179  {
180  if (str1->buf[i] != str2->buf[i])
181  return BFELF_ERROR_MISMATCH;
182 
183  if (str1->buf[i] == 0 ||
184  str2->buf[i] == 0)
185  return BFELF_ERROR_MISMATCH;
186  }
187 
188  return BFELF_SUCCESS;
189 }

195 static bfelf64_sword
196 private_get_string(struct bfelf_file_t *ef,
197  struct bfelf_shdr *strtab,
198  bfelf64_word offset,
199  struct e_string_t *str)
200 {
201  bfelf64_sword max = strtab->sh_size - offset;
202 
203  str->buf = ef->file + strtab->sh_offset + offset;
204  str->len = 0;
205 
206  for (str->len = 0; str->len < max; str->len++)
207  {
208  if (str->buf[str->len] == 0)
209  break;
210  }
211 
212  if (str->len >= max)
213  goto failure;
214 
215  return BFELF_SUCCESS;
216 
217 failure:
218 
219  str->buf = 0;
220  str->len = 0;
221 
222  return invalid_file("the dynamic string table is corrupt");
223 }

277 static bfelf64_sword
278 private_validate_bounds(struct bfelf_file_t *ef)
279 {
280  bfelf64_xword phtab_size = ef->ehdr->e_phoff +
281  (ef->ehdr->e_phentsize * ef->ehdr->e_phnum);
282  bfelf64_xword shtab_size = ef->ehdr->e_shoff +
283  (ef->ehdr->e_shentsize * ef->ehdr->e_shnum);
284 
285  if (ef->ehdr->e_ehsize != sizeof(struct bfelf64_ehdr))
286  return invalid_file("unexpected header size");
287 
288  if (ef->ehdr->e_phentsize != sizeof(struct bfelf_phdr))
289  return invalid_file("unexpected program header size");
290 
291  if (ef->ehdr->e_shentsize != sizeof(struct bfelf_shdr))
292  return invalid_file("unexpected section header size");
293 
294  if (phtab_size > ef->fsize)
295  return invalid_file("corrupt program header table");
296 
297  if (shtab_size > ef->fsize)
298  return invalid_file("corrupt section header table");
299 
300  if (ef->ehdr->e_shstrndx >= ef->ehdr->e_shnum)
301  return invalid_file("section header string table index out of bounds");
302 
303  return BFELF_SUCCESS;
304 }

321 private_get_section_by_name(struct bfelf_file_t *ef,
322  struct e_string_t *name,
323  struct bfelf_shdr **shdr)
324 {
325  bfelf64_sword i = 0;
326  bfelf64_sword ret = 0;
327 
328  for (i = 0, *shdr = 0; i < ef->ehdr->e_shnum; i++)
329  {

332

348 static bfelf64_sword
349 private_check_segments(struct bfelf_file_t *ef)
350 {
351  bfelf64_sword i = 0;
352 
353  for (i = 0; i < ef->ehdr->e_phnum; i++)
354  {
355  struct bfelf_phdr *phdr = private_get_segment(ef, i);
356 
357  if (phdr->p_type != bfpt_load)
358  continue;
359 
360  if (ef->num_loadable_segments >= BFELF_MAX_SEGMENTS)
361  return loader_full("increase BFELF_MAX_SEGMENTS");
362 
363  if (phdr->p_memsz < phdr->p_filesz)
364  return invalid_segment("segment mem size is less then file size");
365 
366  if (phdr->p_vaddr != phdr->p_paddr)
367  return invalid_segment("expect p_vaddr == p_paddr");
368 
369  /*
370  * At the moment, libc++.so has an alignment of 20, which is really
371  * strange. Will need to find out why at some point
372  *
373  * if (phdr->p_align != 0x1000 && phdr->p_align != 0x200000)
374  * return invalid_segment("expect 4k or 2M alignment");
375  */
376 
377  if (phdr->p_offset >= ef->fsize)
378  return invalid_segment("segment offset out of bounds");
379 
380  ef->loadable_segments[ef->num_loadable_segments] = phdr;
381  ef->num_loadable_segments++;
382  }
383 
384  return BFELF_SUCCESS;
385 }

387 static bfelf64_sword
388 private_check_sections(struct bfelf_file_t *ef)
389 {
390  bfelf64_sword i = 0;
391  bfelf64_sword j = 0;
392  struct bfelf_shdr *shstrtab = private_get_section(ef, ef->ehdr->e_shstrndx);
393 
394  for (i = 0; i < ef->ehdr->e_shnum; i++)
395  {
396  bfelf64_sword valid_addr = 0;
397  struct bfelf_shdr *shdr = private_get_section(ef, i);
398 
399  if (shdr->sh_type != bfsht_nobits)
400  {
401  if (shdr->sh_offset + shdr->sh_size > ef->fsize)
402  return invalid_section("section offset / size is corrupt");
403  }
404 
405  if (shdr->sh_name >= shstrtab->sh_size)
406  return invalid_section("invalid section name offset");
407 
408  if (shdr->sh_link >= ef->ehdr->e_shnum)
409  return invalid_section("invalid section link");
410 
411  for (j = 0; j < ef->ehdr->e_phnum; j++)
412  {
413  struct bfelf_phdr *phdr = private_get_segment(ef, j);
414 
415  if (shdr->sh_addr >= phdr->p_vaddr &&
416  shdr->sh_addr + shdr->sh_size <= phdr->p_vaddr + phdr->p_memsz)
417  {
418  valid_addr = 1;
419  break;
420  }
421  }
422 
423  if (shdr->sh_addr != 0 && valid_addr == 0)
424  return invalid_section("section address is out of bounds");
425  }
426 
427  return BFELF_SUCCESS;
428 }

477 private_symbol_table_sections(struct bfelf_file_t *ef)
478 {
479  bfelf64_sword i = 0;
480  bfelf64_sword ret = 0;
481 
482  for (i = 0; i < ef->ehdr->e_shnum; i++)
483  {
484  struct bfelf_shdr *shdr = private_get_section(ef, i);
485 
486  if (shdr->sh_type == bfsht_dynsym)
487  {
488  ret = private_check_section(shdr, bfsht_dynsym, bfshf_a, 8,

491  return ret;

495  }
496

577 static bfelf64_sword
578 private_get_hash_table(struct bfelf_file_t *ef)
579 {
580  if (ef->hashtab)
581  {
582  bfelf64_word total = 0;
583  bfelf64_word *p = (bfelf64_word *)(ef->hashtab->sh_offset + ef->file);
584 
585  if (sizeof(bfelf64_word) * 2 > ef->hashtab->sh_size)
586  return invalid_section("hash table contents corrupt");
587 
588  ef->nbucket = p[0];
589  ef->nchain = p[1];
590 
591  total = (ef->nbucket + ef->nchain + 2) * sizeof(bfelf64_word);
592  if (total > ef->hashtab->sh_size)
593  return invalid_section("hash table contents corrupt");
594 
595  ef->bucket = &(p[2]);
596  ef->chain = &(p[2 + ef->nbucket]);
597  }
598 
599  return BFELF_SUCCESS;
600 }
