77  addr4 = pool.alloc((1 << 3));
78 
79  this->expect_true(addr1 == 100 + ((1 << 3) * 0)); // 100
80  this->expect_true(addr2 == 100 + ((1 << 3) * 1)); // 108
81  this->expect_true(addr3 == 100 + ((1 << 3) * 2)); // 116
82  this->expect_true(addr4 == 100 + ((1 << 3) * 3)); // 124
83 
84  pool.free(addr1);
