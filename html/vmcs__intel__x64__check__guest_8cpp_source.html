229 vmcs_intel_x64::check_guest_verify_load_ia32_efer()
230 {
231  if (is_enabled_load_ia32_efer_on_entry() == false)
232  return;
233 
234  auto efer = vmread(VMCS_GUEST_IA32_EFER_FULL);
235 
236  if ((efer & 0xFFFFFFFFFFFFF2FE) != 0)
237  throw vmcs_invalid_field("ia32 efer msr reserved buts must be 0 if "
238  "load ia32 efer entry is enabled", efer);
239 
240  auto cr0 = vmread(VMCS_GUEST_CR0);
241  auto lma = (efer && IA32_EFER_LMA);
242  auto lme = (efer && IA32_EFER_LME);
243 
244  if (is_enabled_ia_32e_mode_guest() == false && lma != 0)
245  throw vmcs_invalid_field("ia 32e mode is 0, but efer.lma is 1. "
246  "they must be equal", lma);
247 
248  if (is_enabled_ia_32e_mode_guest() == true && lma == 0)
249  throw vmcs_invalid_field("ia 32e mode is 1, but efer.lma is 0. "
250  "they must be equal", lma);
251 
252  if ((cr0 & CR0_PG_PAGING) == 0)
253  return;
254 
255  if (lme == 0 && lma != 0)
256  throw vmcs_invalid_field("efer.lme is 0, but efer.lma is 1. "
257  "they must be equal", lma);
258 
259  if (lme != 0 && lma == 0)
260  throw vmcs_invalid_field("efer.lme is 1, but efer.lma is 0. "
261  "they must be equal", lma);
