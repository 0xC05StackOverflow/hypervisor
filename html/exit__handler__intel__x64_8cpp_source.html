436 void
437 exit_handler_intel_x64::handle_rdmsr()
438 {
439  uint64_t msr = 0;
440 
441  switch (g_guest_rcx)
442  {
443  case IA32_DEBUGCTL_MSR:
444  msr = vmread(VMCS_GUEST_IA32_DEBUGCTL_FULL);
445  break;
446  case IA32_PAT_MSR:
447  msr = vmread(VMCS_GUEST_IA32_PAT_FULL);
448  break;
449  case IA32_EFER_MSR:
450  msr = vmread(VMCS_GUEST_IA32_EFER_FULL);
451  break;
452  case IA32_SYSENTER_CS_MSR:
453  msr = vmread(VMCS_GUEST_IA32_SYSENTER_CS);
454  break;
455  case IA32_SYSENTER_ESP_MSR:
456  msr = vmread(VMCS_GUEST_IA32_SYSENTER_ESP);
457  break;
458  case IA32_SYSENTER_EIP_MSR:
459  msr = vmread(VMCS_GUEST_IA32_SYSENTER_EIP);
460  break;
461  case IA32_FS_BASE_MSR:
462  msr = vmread(VMCS_GUEST_FS_BASE);
463  break;
464  case IA32_GS_BASE_MSR:
465  msr = vmread(VMCS_GUEST_GS_BASE);
466  break;
467  default:
468  break;
469  }
470 
471  switch (g_guest_rcx)
472  {
473  case IA32_DEBUGCTL_MSR:
474  case IA32_PAT_MSR:
475  case IA32_EFER_MSR:
476  case IA32_SYSENTER_CS_MSR:
477  case IA32_SYSENTER_ESP_MSR:
478  case IA32_SYSENTER_EIP_MSR:
479  case IA32_FS_BASE_MSR:
480  case IA32_GS_BASE_MSR:
481  g_guest_rax = ((msr & 0x00000000FFFFFFFF) >> 0);
482  g_guest_rdx = ((msr & 0xFFFFFFFF00000000) >> 32);
483  break;
484  default:
485  m_intrinsics->read_msr_reg(g_guest_rcx, &g_guest_rdx, &g_guest_rax);
486  break;
487  }
488 
489  advance_rip();
490 }

492 void
493 exit_handler_intel_x64::handle_wrmsr()
494 {
495  uint64_t msr = 0;
496 
497  switch (g_guest_rcx)
498  {
499  case IA32_DEBUGCTL_MSR:
500  case IA32_PAT_MSR:
501  case IA32_EFER_MSR:
502  case IA32_SYSENTER_CS_MSR:
503  case IA32_SYSENTER_ESP_MSR:
504  case IA32_SYSENTER_EIP_MSR:
505  case IA32_FS_BASE_MSR:
506  case IA32_GS_BASE_MSR:
507  msr |= ((g_guest_rax << 0) & 0x00000000FFFFFFFF);
508  msr |= ((g_guest_rdx << 32) & 0xFFFFFFFF00000000);
509  break;
510  default:
511  break;
512  }
513 
514  switch (g_guest_rcx)
515  {
516  case IA32_DEBUGCTL_MSR:
517  vmwrite(VMCS_GUEST_IA32_DEBUGCTL_FULL, msr);
518  break;
519  case IA32_PAT_MSR:
520  vmwrite(VMCS_GUEST_IA32_PAT_FULL, msr);
521  break;
522  case IA32_EFER_MSR:
523  vmwrite(VMCS_GUEST_IA32_EFER_FULL, msr);
524  break;
525  case IA32_SYSENTER_CS_MSR:
526  vmwrite(VMCS_GUEST_IA32_SYSENTER_CS, msr);
527  break;
528  case IA32_SYSENTER_ESP_MSR:
529  vmwrite(VMCS_GUEST_IA32_SYSENTER_ESP, msr);
530  break;
531  case IA32_SYSENTER_EIP_MSR:
532  vmwrite(VMCS_GUEST_IA32_SYSENTER_EIP, msr);
533  break;
534  case IA32_FS_BASE_MSR:
535  vmwrite(VMCS_GUEST_FS_BASE, msr);
536  break;
537  case IA32_GS_BASE_MSR:
538  vmwrite(VMCS_GUEST_GS_BASE, msr);
539  break;
540  default:
541  m_intrinsics->write_msr_reg(g_guest_rcx, g_guest_rdx, g_guest_rax);
542  break;
543  }
544 
545  advance_rip();
546 }
