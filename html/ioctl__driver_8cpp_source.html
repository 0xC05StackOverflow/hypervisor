91 ioctl_driver::load_vmm(const std::shared_ptr<file> &f,
92  const std::shared_ptr<ioctl> &ctl,
93  const std::shared_ptr<command_line_parser> &clp)
94 {
95  switch (get_status(ctl))
96  {
97  case VMM_RUNNING:
98  stop_vmm(ctl);
99 
100  case VMM_LOADED:
101  case VMM_UNLOADED:
102  unload_vmm(ctl);
103  break;
104 
105  case VMM_CORRUPT:
106  throw corrupt_vmm();
107 
108  default:
109  throw unknown_status();
110  }
111 
112  auto cor1 = commit_or_rollback([&]
113  { unload_vmm(ctl); });
114 
115  for (const auto &module : split(f->read(clp->modules()), '\n'))
116  {
117  auto trimmed = trim(module);
118 
119  if (trimmed.empty() == true)
120  continue;
121 
122  ctl->call_ioctl_add_module(f->read(trimmed));
123  }
124 
125  ctl->call_ioctl_load_vmm();
126 
127  cor1.commit();
128 }
