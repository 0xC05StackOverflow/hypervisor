47 vmcs_intel_x64::check_host_cr0_for_unsupported_bits()
48 {
49  auto cr0 = vmread(VMCS_HOST_CR0);
50  auto ia32_vmx_cr0_fixed0 = m_intrinsics->read_msr(IA32_VMX_CR0_FIXED0_MSR);
51  auto ia32_vmx_cr0_fixed1 = m_intrinsics->read_msr(IA32_VMX_CR0_FIXED1_MSR);
52 
53  if (0 != ((~cr0 & ia32_vmx_cr0_fixed0) | (cr0 & ~ia32_vmx_cr0_fixed1)))
54  throw vmcs_invalid_ctrl(cr0, ia32_vmx_cr0_fixed0, ia32_vmx_cr0_fixed1);
55 }

149 void
150 vmcs_intel_x64::check_host_verify_load_ia32_efer()
151 {
152  if (is_enabled_load_ia32_efer_on_exit() == false)
153  return;
154 
155  auto efer = vmread(VMCS_HOST_IA32_EFER_FULL);
156 
157  if ((efer & 0xFFFFFFFFFFFFF2FE) != 0)
158  throw vmcs_invalid_field("ia32 efer msr reserved buts must be 0 if "
159  "load ia32 efer entry is enabled", efer);
160 
161  auto cr0 = vmread(VMCS_HOST_CR0);
162  auto lma = (efer && IA32_EFER_LMA);
163  auto lme = (efer && IA32_EFER_LME);
164 
165  if (is_enabled_host_address_space_size() == false && lma != 0)
166  throw vmcs_invalid_field("host addr space is 0, but efer.lma is 1. "
167  "they must be equal", lma);
168 
169  if (is_enabled_host_address_space_size() == true && lma == 0)
170  throw vmcs_invalid_field("host addr space is 1, but efer.lma is 0. "
171  "they must be equal", lma);
172 
173  if ((cr0 & CR0_PG_PAGING) == 0)
174  return;
175 
176  if (lme == 0 && lma != 0)
177  throw vmcs_invalid_field("efer.lme is 0, but efer.lma is 1. "
178  "they must be equal", lma);
179 
180  if (lme != 0 && lma == 0)
181  throw vmcs_invalid_field("efer.lme is 1, but efer.lma is 0. "
182  "they must be equal", lma);
183 }
