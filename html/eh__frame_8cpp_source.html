31 uint64_t
32 decode_pointer(char **addr, uint64_t encoding)
33 {
34  uint64_t result = 0;
35 
36  if (encoding == DW_EH_PE_omit)
37  return 0;
38 
39  // The following are defined in the DWARF Exception Header Encodings
40  // section 10.5.1. For some reason, GCC adds a 0x80 to the upper 4 bits
41  // that are not documented in the LSB. Thefore, only 3 of the upper 4 bits
42  // are actually used.
43 
44  switch (encoding & 0x70)
45  {
46  case DW_EH_PE_absptr:
47  break;
48 
49  case DW_EH_PE_pcrel:
50  result += (uint64_t) * addr;
51  break;
52 
53  case DW_EH_PE_textrel:
54  ABORT("DW_EH_PE_textrel pointer encodings not supported");
55  break;
56 
57  case DW_EH_PE_datarel:
58  ABORT("DW_EH_PE_datarel pointer encodings not supported");
59  break;
60 
61  case DW_EH_PE_funcrel:
62  ABORT("DW_EH_PE_funcrel pointer encodings not supported");
63  break;
64 
65  case DW_EH_PE_aligned:
66  ABORT("DW_EH_PE_aligned pointer encodings not supported");
67  break;
68 
69  default:
70  ABORT("unknown upper pointer encoding bits");
71  }
72 
73  switch (encoding & 0x0F)
74  {
75  case DW_EH_PE_absptr:
76  result += (uint64_t) * (void **)*addr;
77  *addr += sizeof(void *);
78  break;
79 
80  case DW_EH_PE_uleb128:
81  result += (uint64_t)dwarf4::decode_uleb128(addr);
82  break;
83 
84  case DW_EH_PE_udata2:
85  result += (uint64_t) * (uint16_t *)*addr;
86  *addr += sizeof(uint16_t);
87  break;
88 
89  case DW_EH_PE_udata4:
90  result += (uint64_t) * (uint32_t *)*addr;
91  *addr += sizeof(uint32_t);
92  break;
93 
94  case DW_EH_PE_udata8:
95  result += (uint64_t) * (uint64_t *)*addr;
96  *addr += sizeof(uint64_t);
97  break;
98 
99  case DW_EH_PE_sleb128:
100  result += (uint64_t)dwarf4::decode_sleb128(addr);
101  break;
102 
103  case DW_EH_PE_sdata2:
104  result += (uint64_t) * (int16_t *)*addr;
105  *addr += sizeof(int16_t);
106  break;
107 
108  case DW_EH_PE_sdata4:
109  result += (uint64_t) * (int32_t *)*addr;
110  *addr += sizeof(int32_t);
111  break;
112 
113  case DW_EH_PE_sdata8:
114  result += (uint64_t) * (int64_t *)*addr;
115  *addr += sizeof(int64_t);
116  break;
117 
118  default:
119  ABORT("unknown lower pointer encoding bits");
120  }
121 
122  return result;
123 }

162 void
163 common_entry::parse(char *addr)
164 {
165  auto len = 0ULL;
166 
167  if ((m_entry_start = addr) == 0)
168  goto failure;
169 
170  if (m_entry_start < m_eh_frame.addr)
171  goto failure;
172 
173  if (((uint32_t *)(m_entry_start))[0] != 0xFFFFFFFF)
174  {
175  len = ((uint32_t *)(m_entry_start + 0))[0];
176  m_payload_start = m_entry_start + 4;
177  }
178  else
179  {
180  len = ((uint64_t *)(m_entry_start + 4))[0];
181  m_payload_start = m_entry_start + 12;
182  }
183 
184  if (len == 0)
185  goto failure;
186 
187  m_payload_end = m_payload_start + len;
188  m_entry_end = m_payload_end;
189 
190  if (m_entry_end > (char *)m_eh_frame.addr + m_eh_frame.size)
191  goto failure;
192 
193  m_is_cie = (((uint32_t *)(m_payload_start))[0] == 0);
194  return;
195 
196 failure:
197 
198  m_is_cie = false;
199  m_entry_start = 0;
200  m_entry_end = 0;
201  m_payload_start = 0;
202  m_payload_end = 0;
203 }

254 void
255 ci_entry::parse(char *addr)
256 {
257  common_entry::parse(addr);
258 
259  if (*this == false)
260  return;
261 
262  if (is_cie() == false)
263  return;
264 
265  auto p = payload_start();
266 
267  p += sizeof(uint32_t);
268  p += sizeof(uint8_t);
269 
270  m_augmentation_string = p;
271 
272  while (*p++ != 0);
273 
274  m_code_alignment = dwarf4::decode_uleb128(&p);
275  m_data_alignment = dwarf4::decode_sleb128(&p);
276  m_return_address_reg = dwarf4::decode_uleb128(&p);
277 
278  if (m_augmentation_string[0] == 'z')
279  {
280  auto len = dwarf4::decode_uleb128(&p);
281 
282  for (auto i = 1U; m_augmentation_string[i] != 0 && i <= len; i++)
283  {
284  switch (m_augmentation_string[i])
285  {
286  case 'L':
287  m_lsda_encoding = *(uint8_t *)p++;
288  break;
289 
290  case 'P':
291  m_personality_encoding = *(uint8_t *)p++;
292  m_personality_function =
293  decode_pointer(&p, m_personality_encoding);
294  break;
295 
296  case 'R':
297  m_pointer_encoding = *(uint8_t *)p++;
298  break;
299 
300  default:
301  ABORT("unknown augmentation string character");
302  }
303  }
304  }
305 
306  m_initial_instructions = p;
307 }

343 void
344 fd_entry::parse(char *addr)
345 {
346  common_entry::parse(addr);
347 
348  if (*this == false)
349  return;
350 
351  if (is_fde() == false)
352  return;
353 
354  auto p = payload_start();
355  auto p_cie = (char *)((uint64_t)p - * (uint32_t *)p);
356 
357  m_cie = ci_entry(eh_frame(), p_cie);
358  p += sizeof(uint32_t);
359 
360  m_pc_begin = decode_pointer(&p, m_cie.pointer_encoding());
361  m_pc_range = decode_pointer(&p, m_cie.pointer_encoding() & 0xF);
362 
363  if (m_cie.augmentation_string(0) == 'z')
364  {
365  auto len = dwarf4::decode_uleb128(&p);
366 
367  for (auto i = 1U; m_cie.augmentation_string(i) != 0 && i <= len; i++)
368  {
369  switch (m_cie.augmentation_string(i))
370  {
371  case 'L':
372  m_lsda = decode_pointer(&p, m_cie.lsda_encoding());
373  break;
374 
375  case 'P':
376  break;
377 
378  case 'R':
379  break;
380 
381  default:
382  ABORT("unknown augmentation string character");
383  }
384  }
385  }
386 
387  m_instructions = p;
